-- โครงสร้าง build.gradle 
apply plugin: 'com.android.application'   // เป็นการ บอกว่า mudule ตัวนี้ เป็น application หรือ  library 

android {
    compileSdkVersion 25 // เป็นการบอกว่า เราใข้ sdk ไหนมา complie 
    buildToolsVersion "25.0.2"  // คือเลขของ Build Tools 
    defaultConfig {
        applicationId "com.mis_school.helloworld" // เป็นการระบุว่า app เราคือ id อะไร
        minSdkVersion 15  // เป็นการระบุว่า app เราจะ run ต่ำสุดคือ Ver 
        targetSdkVersion 25 // app ของเรา ให้ใช้ได้ถึง api V.25 อย่าปรับเองดื้อ ๆ
        versionCode 1 // ver ของโปรแกรม 
        versionName "1.0" // ver name
        testInstrumentationRunner "android.support.test.runner.AndroidJUnitRunner" 
    }
    buildTypes {
        release {
            minifyEnabled false
            proguardFiles getDefaultProguardFile('proguard-android.txt'), 'proguard-rules.pro'
        }
    }
}

// การที่ระบุ
dependencies {
    compile fileTree(dir: 'libs', include: ['*.jar'])
    androidTestCompile('com.android.support.test.espresso:espresso-core:2.2.2', {
        exclude group: 'com.android.support', module: 'support-annotations'
    })
    compile 'com.android.support:appcompat-v7:25.2.0'
    testCompile 'junit:junit:4.12'
}

- match_parent (fill_parent) อิงกับแม่ คือขยายให้เต็ม fill_parent ไม่แนะนำให้ใช้คำนี้
